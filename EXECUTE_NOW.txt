STEP 11: VERIFY ALL 16 ENDPOINTS ARE WORKING (Copy entire block)
================================================================================

bash << 'ENDPOINTS'
echo "🔍 TESTING ALL 16 API ENDPOINTS"
echo "================================"

ENDPOINTS=(
  "POST:/api/accuracy/enhance"
  "POST:/api/accuracy/planning"
  "POST:/api/accuracy/risk-assessment"
  "POST:/api/accuracy/testing"
  "POST:/api/accuracy/reconciliation"
  "POST:/api/accuracy/reporting"
  "POST:/api/accuracy/anomalies"
  "POST:/api/accuracy/fraud-detection"
  "POST:/api/accuracy/data-quality"
  "POST:/api/accuracy/confidence-scores"
  "POST:/api/accuracy/consensus"
  "POST:/api/accuracy/export"
  "GET:/api/accuracy/metrics"
  "GET:/api/accuracy/status"
  "GET:/health"
  "GET:/"
)

for endpoint in "${ENDPOINTS[@]}"; do
  METHOD=$(echo $endpoint | cut -d: -f1)
  PATH=$(echo $endpoint | cut -d: -f2)

  if [ "$METHOD" = "POST" ]; then
    RESULT=$(curl -s -X POST http://localhost:3000${PATH} \
      -H "Content-Type: application/json" \
      -d '{}' 2>/dev/null | jq -r '.success // .status // .message // "RESPONSE"' 2>/dev/null)
  else
    RESULT=$(curl -s http://localhost:3000${PATH} 2>/dev/null | jq -r '.status // .message // "RESPONSE"' 2>/dev/null)
  fi

  if [ ! -z "$RESULT" ] && [ "$RESULT" != "null" ]; then
    echo "✅ $METHOD $PATH"
  else
    echo "⚠️  $METHOD $PATH (waiting for startup...)"
  fi
done

echo ""
echo "================================"
echo "✅ All endpoints test complete"
ENDPOINTS

Expected: Most endpoints should show ✅ status
